import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.decomposition import LatentDirichletAllocation

from sklearn.feature_extraction import text

# Read the positive reviews data
positive_reviews = pd.read_csv("C:\\Users\\ASUS\\Downloads\\WEEK3 new\\WEEK3\\Food & Beverages\\food refresh1 - Copy.csv")

# Combine reviews into a list
reviews = positive_reviews['tweet'].tolist()

# Create a custom stop words list if needed
custom_stop_words = ["read", "good", "disney", "yorker", "day", "didn", "stay", "great", "check", "booked", "like", "die", "told", "friendly", "clean", "sehr", "das", "zimmer", "und", "nice", "did", "was", "really", "safe", "walk", "close", "got"]

# Extend the default English stop words list with custom stop words
stop_words = list(text.ENGLISH_STOP_WORDS) + custom_stop_words

# Vectorize the text data
vectorizer = CountVectorizer(max_df=0.95, min_df=2, stop_words=stop_words)
X = vectorizer.fit_transform(reviews)

# Apply LDA for topic modeling
num_topics = 5
lda = LatentDirichletAllocation(n_components=num_topics, random_state=42)
lda.fit(X)

# Extract top words for each topic
feature_names = vectorizer.get_feature_names_out()
topics = []
for topic_idx, topic in enumerate(lda.components_):
    top_words_idx = topic.argsort()[-10:]
    top_words = [feature_names[i] for i in top_words_idx]
    topics.append(top_words)

# Print the top words for each topic (possible reasons for positive reviews)
for i, topic in enumerate(topics):
    print(f"Topic {i + 1}: {', '.join(topic)}")
